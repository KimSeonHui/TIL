function solution(new_id) {

            .match(/[\w\-\.]/g).join('') //2
            .replace(/\.{2,}/g, '.') //3
            .replace(/^\.|\.$/g, '') //4
            .replace(/^$/, 'a') //5
    
    //6
    if(answer.length >= 16) {  
        answer = answer.slice(0, 15);
        if (answer[answer.length - 1] === '.') {
            answer = answer.slice(0, answer.length - 1);
        } 
    }
    
    //7
    else if(answer.length <= 2) {
        while(answer.length < 3) {
            answer += answer[answer.length - 1];
        }
    }  
    return answer;
}
